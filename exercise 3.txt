CREATE TABLE `university`.`course` (
    `course_no` INT NOT NULL,
    `course_name` VARCHAR(100) NOT NULL,
    `credit_hours` INT NOT NULL,
    `department` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`course_no`)
);
CREATE TABLE `university`.`section` (
    `section_no` INT NOT NULL,
    `course_no` INT NOT NULL,
    `semester` VARCHAR(100) NOT NULL,
	`year` INT NOT NULL,
	`instructor` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`section_no`),
	FOREIGN KEY (`course_no`) REFERENCES course(`course_no`)
);
DROP TABLE university.section;
CREATE TABLE `university`.`gradereport` (
    `student_no` INT NOT NULL,
    `section_no` INT NOT NULL,
    `grade` VARCHAR(10) NOT NULL,
    FOREIGN KEY (`student_no`) REFERENCES stud(`student_no`),
    FOREIGN KEY (`section_no`) REFERENCES section(`section_no`)
    );
CREATE TABLE `university`.`prerequisite`(
`course_no` INT NOT NULL,
`prereq_no` INT NOT NULL,
FOREIGN KEY (`course_no`) REFERENCES course(`course_no`)
);

SELECT c.course_name, g.grade
FROM university.stud s
JOIN university.gradereport g ON s.student_no = g.student_no
JOIN university.section se ON g.section_no = se.section_no                                       #1
JOIN university.course c ON se.course_no = c.course_no
WHERE s.name = 'smith';

SELECT s.name, g.grade
FROM university.stud s
JOIN university.gradereport g ON s.student_no = g.student_no
JOIN university.section se ON g.section_no = se.section_no
JOIN university.course c ON se.course_no = c.course_no
WHERE c.course_name = 'database'                                                                      #2
AND se.semester = 'fall'
AND se.year = 8;

SELECT p.prereq_no
FROM university.prerequisite p
JOIN university.course c ON p.course_no = c.course_no                                                    #3
WHERE c.course_name = 'database';

SELECT DISTINCT c.course_name
FROM university.course c
JOIN university.section s ON c.course_no = s.course_no                                                                #5
WHERE s.instructor = 'king'
AND s.year IN (7,8);

SELECT s.course_no, s.semester, s.year,
COUNT(g.student_no) AS total_students                                                                        #6
FROM university.section s
LEFT JOIN university.gradereport g
ON s.section_no = g.section_no
WHERE s.instructor = 'king'
GROUP BY s.section_no;

SELECT s.name, c.course_name, c.course_no,
c.credit_hours, se.semester, se.year, g.grade
FROM university.stud s                                                                                              #7
JOIN university.gradereport g ON s.student_no = g.student_no
JOIN university.section se ON g.section_no = se.section_no
JOIN university.course c ON se.course_no = c.course_no
WHERE s.class = 4 AND s.major = 'cs'
ORDER BY s.name;
